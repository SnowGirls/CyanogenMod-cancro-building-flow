************ 源码方式 让子 project 查找 project ************

settings.gradle
// :vision-camera 不打包 aar 了用源码依赖的方式
include ':vision-camera'
rootProject.children.each { proj ->
    if (proj.name == "flutter_vision_camera") {
        println "🍀🍀🍀 Making source dependency(NOT .AAR) for project :vision-camera on path: ${proj.projectDir}/../../android/vision-camera\n"
        project(':vision-camera').projectDir = new File(proj.projectDir, '../../android/vision-camera')
    }
}

// 打印一下所有的全局变量与Project属性，便于调试
println "================ Project Properties ================"
project.properties.each { prop ->
     println "${prop.key} = ${prop.value}"
    // 如果 key 为 flutter 的话，可以打印更多信息
    if (prop.key == 'flutter') {
        // 打印 flutter 所有属性
        flutter.properties.each { fProp ->
            println "    ${fProp.key} = ${fProp.value}"
        }
    }
}


************ 打印项目里的 所有库及其对应的maven仓库地址 ************
tasks.register("printRepos") {
    doLast {
        println("=== Repositories ===")
        allprojects.forEach { p ->
            p.repositories.forEach { r ->
                try {
                    val url = when (r) {
                        is org.gradle.api.artifacts.repositories.MavenArtifactRepository -> r.url
                        else -> null
                    }
                    println("${p.path} -> ${r.name} : ${url ?: "-"}")
                } catch (e: Exception) {
                    println("${p.path} -> ${r.name} : (error: ${e.message})")
                }
            }
        }
    }
}



plugins.withId("com.android.library") {
    val libExt = extensions.getByType<LibraryExtension>()
    if (libExt.namespace == null || libExt.namespace!!.isBlank()) {
        val manifestFile = file("${projectDir}/src/main/AndroidManifest.xml")
        if (manifestFile.exists()) {
            val manifestContent = manifestFile.readText()
            val packageName = Regex("""package\s*=\s*["']([^"']+)["']""")
                .find(manifestContent)
                ?.groupValues?.getOrNull(1)
            if (!packageName.isNullOrBlank()) {
                println("✓ 为插件 ${project.name} 注入 namespace: $packageName")
                libExt.namespace = packageName
            }
        }
    }
}
